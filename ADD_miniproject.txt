document.addEventListener("DOMContentLoaded", () => {

    /* ================= DARK MODE BUTTON ================= */
    const darkBtn = document.createElement("button");
    darkBtn.id = "darkBtn";
    darkBtn.className = "floating-btn";
    document.body.appendChild(darkBtn);

    function updateDarkBtn() {
        darkBtn.innerText = document.body.classList.contains("dark") ? "☀" : "🌙";
    }

    // load saved theme
    if (localStorage.getItem("theme") === "dark") {
        document.body.classList.add("dark");
    }

    updateDarkBtn();

    darkBtn.addEventListener("click", () => {
        document.body.classList.toggle("dark");

        localStorage.setItem(
            "theme",
            document.body.classList.contains("dark") ? "dark" : "light"
        );

        updateDarkBtn();
    });

    /* ================= TOAST ================= */
    let currentToast;

    function toast(msg) {
        if (currentToast) currentToast.remove();

        const t = document.createElement("div");
        currentToast = t;

        t.innerText = msg;
        t.style.position = "fixed";
        t.style.bottom = "80px";
        t.style.left = "50%";
        t.style.transform = "translateX(-50%)";
        t.style.background = "#2e8b57";
        t.style.color = "white";
        t.style.padding = "12px 18px";
        t.style.borderRadius = "8px";
        t.style.zIndex = "9999";

        document.body.appendChild(t);

        setTimeout(() => t.remove(), 2500);
    }

    /* ================= FORM ================= */
    const form = document.querySelector("form");

    if (form) {
        form.addEventListener("submit", (e) => {
            e.preventDefault();

            const name = document.querySelector("#name").value.trim();
            const email = document.querySelector("#email").value.trim();
            const phone = document.querySelector("#phone").value.trim();

            if (!name || !email || !phone) {
                toast("⚠ Please fill all required fields");
                return;
            }

            const phonePattern = /^[0-9]{10}$/;
            if (!phonePattern.test(phone)) {
                toast("📵 Enter valid 10-digit phone number");
                return;
            }

            toast("🎉 Booking submitted successfully!");
            form.reset();
        });
    }

    /* ================= BACK TO TOP BUTTON ================= */
    const topBtn = document.createElement("button");
    topBtn.id = "topBtn";
    topBtn.className = "floating-btn";
    topBtn.innerText = "⬆";
    topBtn.style.display = "none";

    document.body.appendChild(topBtn);

    window.addEventListener("scroll", () => {
        topBtn.style.display = window.scrollY > 300 ? "flex" : "none";
    });

    topBtn.addEventListener("click", () => {
        window.scrollTo({ top: 0, behavior: "smooth" });
    });

});
